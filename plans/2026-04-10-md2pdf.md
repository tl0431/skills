# md2pdf — Markdown to PDF Converter

## What this skill does

Converts a Markdown file to a professionally styled PDF using pure Python (ReportLab). No browser required. Full Chinese support. Configurable themes and fonts.

## Language behavior

All prompts are bilingual. Detect the language of the user's most recent message:
- Chinese last → Chinese first, English in parentheses: `请选择字体 / Please select a font:`
- English last → English first, Chinese in parentheses: `Please select a font / 请选择字体:`

## How to run this skill

### Step 1: Resolve input and output paths

Extract the input `.md` file path from the user's message. If relative, resolve to absolute using the current working directory.

Output path defaults to same directory, same name, `.pdf` extension. If the user specified a different output path, use that.

### Step 2: Font selection

Run: `python md2pdf/scripts/font_finder.py --yaml <path-to-pdf_style.yaml>`

This script:
1. Reads `pdf_style.yaml` `fonts` cache — if populated, skips glob scan
2. If no cache: scans system font directories, writes results to yaml
3. Returns JSON: `{"fonts": {"Name": "/path"}, "default_font": "Name or null"}`

If `default_font` is set in yaml → use it, skip prompt.

If not set → show font selection prompt (see format below), ask user to pick.
After user picks, ask: "设为默认字体？/ Set as default font? (y/n)"
- y → write `default_font` to yaml. If this is the FIRST time setting a default, show tip:
  > 已将 [字体名] 设为默认字体。如需修改，可在 `pdf_style.yaml` 中更改 `default_font` 字段，或在调用时说"这次用别的字体"临时覆盖。
  > Default font set to [font name]. To change it, edit `default_font` in `pdf_style.yaml`, or say "use a different font this time" to override temporarily.
- n → use for this session only

**Font selection prompt format:**
```
请选择字体 / Please select a font:
  1  PingFang SC
  2  Arial Unicode MS
  3  STHeiti
  ...
  0  其他 / Other (specify font name)
```

If user picks "Other" and the font is not found on system:
- Ask: "用途是商业还是非商业？/ Commercial or non-commercial use? (commercial/non-commercial)"
- non-commercial → use WebSearch to find a free download link, present top 3 options
- commercial → reply: "请自行购买字体授权并将字体文件放置到系统字体目录后重新运行。/ Please purchase a font license and place the font file in your system fonts directory, then re-run."

Final fallback (no font found at all): use bundled `md2pdf/assets/fonts/NotoSansSC-Regular.ttf`

### Step 3: Theme selection

Read `pdf_style.yaml` `theme` field.
- Set → use it, skip prompt
- Not set → show ANSI theme selector (see format in references/style_schema.md), ask user to pick
- After pick, ask: "设为默认主题？/ Set as default theme? (y/n)"
  - y → write `theme` to yaml
  - n → use for this session only

### Step 4: Run conversion

```bash
python md2pdf/scripts/md2pdf.py \
  --input "<input.md>" \
  --output "<output.pdf>" \
  --font "<font_path>" \
  --theme "<theme_name>" \
  --style "<pdf_style.yaml>"
```

### Step 5: Report result

Tell the user the output path and file size.
Example: `完成 / Done: report.pdf (245 KB)`
```

```python
#!/usr/bin/env python3
"""Font discovery, caching, and selection for md2pdf."""

import glob
import json
import os
import platform
import sys
from pathlib import Path

import yaml

CHINESE_KEYWORDS = [
    "arial unicode", "pingfang", "yahei", "simsun", "simhei",
    "stheiti", "hiragino", "noto cjk", "notosanscjk", "source han",
    "wenquanyi", "notosanssc", "noto sans sc",
]

SCAN_PATHS = {
    "Darwin": [
        "/Library/Fonts/**",
        str(Path.home() / "Library/Fonts/**"),
        "/System/Library/Fonts/**",
    ],
    "Windows": ["C:\\Windows\\Fonts\\**"],
    "Linux": [
        "/usr/share/fonts/**",
        str(Path.home() / ".fonts/**"),
        "/usr/local/share/fonts/**",
    ],
}


def bundled_font_path() -> str:
    """Return absolute path to the bundled NotoSansSC font."""
    here = Path(__file__).parent.parent
    return str(here / "assets" / "fonts" / "NotoSansSC-Regular.ttf")


def scan_system_fonts() -> dict:
    """Scan system font directories. Returns {display_name: path}."""
    system = platform.system()
    patterns = SCAN_PATHS.get(system, SCAN_PATHS["Linux"])
    found = {}
    extensions = ("*.ttf", "*.otf", "*.ttc")
    for base_pattern in patterns:
        for ext in extensions:
            # base_pattern already ends with **
            pattern = base_pattern.rstrip("*").rstrip("/") + "/**/" + ext
            for path in glob.glob(pattern, recursive=True):
                name = Path(path).stem
                found[name] = path
    return found


def filter_chinese_fonts(fonts: dict) -> dict:
    """Keep only fonts whose name matches Chinese font keywords."""
    result = {}
    for name, path in fonts.items():
        lower = name.lower()
        if any(kw in lower for kw in CHINESE_KEYWORDS):
            result[name] = path
    return result


def read_yaml_cache(yaml_path: str) -> dict:
    """Read pdf_style.yaml and return its contents as dict. Returns {} if missing."""
    p = Path(yaml_path)
    if not p.exists():
        return {}
    with open(p, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}
    return data


def write_yaml_cache(yaml_path: str, fonts: dict, default_font: str | None) -> None:
    """Merge fonts cache and optional default_font into pdf_style.yaml."""
    p = Path(yaml_path)
    data = read_yaml_cache(yaml_path)
    if fonts:
        existing = data.get("fonts", {})
        existing.update(fonts)
        data["fonts"] = existing
    if default_font is not None:
        data["default_font"] = default_font
    with open(p, "w", encoding="utf-8") as f:
        yaml.dump(data, f, allow_unicode=True, default_flow_style=False)


def get_fonts(yaml_path: str) -> tuple[dict, str | None]:
    """
    Return (chinese_fonts_dict, default_font_or_None).
    Uses yaml cache if available; otherwise scans system and writes cache.
    """
    data = read_yaml_cache(yaml_path)
    cached = data.get("fonts", {})
    default = data.get("default_font")

    if cached:
        return cached, default

    # No cache — scan system
    all_fonts = scan_system_fonts()
    chinese = filter_chinese_fonts(all_fonts)

    # Add bundled font as fallback entry
    bundled = bundled_font_path()
    if Path(bundled).exists():
        chinese["NotoSansSC (bundled)"] = bundled

    write_yaml_cache(yaml_path, chinese, default_font=None)
    return chinese, default


def main():
    """CLI: python font_finder.py --yaml /path/to/pdf_style.yaml"""
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--yaml", required=True, help="Path to pdf_style.yaml")
    args = parser.parse_args()

    fonts, default_font = get_fonts(args.yaml)
    print(json.dumps({"fonts": fonts, "default_font": default_font}, ensure_ascii=False))


if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""Convert Markdown to PDF using ReportLab. Full Chinese support."""

import argparse
import re
import sys
from pathlib import Path

import yaml
from reportlab.lib.colors import HexColor, white
from reportlab.lib.enums import TA_CENTER, TA_JUSTIFY, TA_LEFT
from reportlab.lib.pagesizes import A4, LETTER
from reportlab.lib.styles import ParagraphStyle
from reportlab.lib.units import cm
from reportlab.pdfbase import pdfmetrics
from reportlab.pdfbase.ttfonts import TTFont
from reportlab.platypus import (
    BaseDocTemplate, Frame, HRFlowable, PageBreak,
    PageTemplate, Paragraph, Spacer, Table, TableStyle,
)

THEMES = {
    "navy":     {"accent": "#1C3A5E", "dark": "#1A1A2E", "muted": "#888888"},
    "forest":   {"accent": "#2D5A27", "dark": "#1A2E1A", "muted": "#888888"},
    "minimal":  {"accent": "#333333", "dark": "#111111", "muted": "#999999"},
    "warm":     {"accent": "#7B4F2E", "dark": "#2E1A0E", "muted": "#999999"},
    "coral":    {"accent": "#C0392B", "dark": "#2C1810", "muted": "#888888"},
    "slate":    {"accent": "#4A5568", "dark": "#1A202C", "muted": "#888888"},
    "purple":   {"accent": "#553C9A", "dark": "#1A0E2E", "muted": "#888888"},
    "teal":     {"accent": "#2C7A7B", "dark": "#0E2E2E", "muted": "#888888"},
    "gold":     {"accent": "#B7791F", "dark": "#2E1E0E", "muted": "#888888"},
    "rose":     {"accent": "#B83280", "dark": "#2E0E1E", "muted": "#888888"},
    "midnight": {"accent": "#2D3748", "dark": "#0A0A0A", "muted": "#999999"},
    "olive":    {"accent": "#556B2F", "dark": "#1A2010", "muted": "#888888"},
}

STYLE_DEFAULTS = {
    "page_size": "A4",
    "margin_cm": 2.5,
    "cover": True,
    "cover_title": "",
    "cover_subtitle": "",
    "cover_meta": "",
    "header": True,
    "footer": True,
    "footer_page_number": True,
}


def get_theme(name: str, accent: str = None, dark: str = None, muted: str = None) -> dict:
    """Return theme color dict. Falls back to navy for unknown names."""
    if name == "custom":
        return {
            "accent": accent or "#1C3A5E",
            "dark": dark or "#1A1A2E",
            "muted": muted or "#888888",
        }
    return THEMES.get(name, THEMES["navy"])


def load_style(yaml_path: str) -> dict:
    """Load pdf_style.yaml, filling in defaults for missing keys."""
    style = dict(STYLE_DEFAULTS)
    p = Path(yaml_path)
    if p.exists():
        with open(p, "r", encoding="utf-8") as f:
            data = yaml.safe_load(f) or {}
        for k, v in data.items():
            if k in style:
                style[k] = v
    return style
```

def escape_xml(s: str) -> str:
    return s.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")


def md_inline(s: str) -> str:
    """Convert inline markdown to ReportLab XML tags."""
    s = escape_xml(s)
    s = re.sub(r'\*\*(.+?)\*\*', r'<b>\1</b>', s)
    s = re.sub(r'\*(.+?)\*', r'<i>\1</i>', s)
    s = re.sub(r'`(.+?)`', r'<font name="Courier">\1</font>', s)
    return s


def parse_md(text: str) -> list:
    """Parse markdown into list of (type, content) tuples."""
    blocks = []
    lines = text.splitlines()
    i = 0
    table_buf = []

    def flush_table():
        if not table_buf:
            return
        rows = []
        for row in table_buf:
            cells = [c.strip() for c in row.strip().strip("|").split("|")]
            rows.append(cells)
        rows = [r for r in rows if not all(re.match(r'^[-:]+$', c) for c in r)]
        blocks.append(("table", rows))
        table_buf.clear()

    while i < len(lines):
        line = lines[i]

        if line.strip().startswith("|"):
            table_buf.append(line)
            i += 1
            continue
        else:
            flush_table()

        if not line.strip():
            i += 1
            continue

        m = re.match(r'^(#{1,3})\s+(.*)', line)
        if m:
            blocks.append((f"h{len(m.group(1))}", m.group(2).strip()))
            i += 1
            continue

        if line.strip().startswith("```"):
            code_lines = []
            i += 1
            while i < len(lines) and not lines[i].strip().startswith("```"):
                code_lines.append(lines[i])
                i += 1
            i += 1
            blocks.append(("code", "\n".join(code_lines)))
            continue

        if re.match(r'^---+$', line.strip()):
            blocks.append(("hr", ""))
            i += 1
            continue

        m = re.match(r'^[-*]\s+(.*)', line)
        if m:
            blocks.append(("bullet", m.group(1).strip()))
            i += 1
            continue

        m = re.match(r'^\d+\.\s+(.*)', line)
        if m:
            blocks.append(("num", m.group(1).strip()))
            i += 1
            continue

        m = re.match(r'^>\s*(.*)', line)
        if m:
            blocks.append(("callout", m.group(1).strip()))
            i += 1
            continue

        blocks.append(("body", line.strip()))
        i += 1

    flush_table()
    return blocks
```

def register_font(font_path: str) -> str:
    """Register font with ReportLab. Returns the font name to use."""
    name = "MD2PDF_FONT"
    pdfmetrics.registerFont(TTFont(name, font_path))
    pdfmetrics.registerFont(TTFont(name + "-Bold", font_path))  # same file, no separate bold
    return name


def make_styles(font_name: str, theme: dict) -> dict:
    ACCENT = HexColor(theme["accent"])
    DARK = HexColor(theme["dark"])
    MUTED = HexColor(theme["muted"])
    BG_ROW = HexColor("#F0F4F8")
    BG_CALL = HexColor("#EEF3F8")

    def S(name, **kw):
        base = dict(fontName=font_name, fontSize=10.5, leading=17,
                    textColor=DARK, spaceAfter=6)
        base.update(kw)
        return ParagraphStyle(name, **base)

    return {
        "h1":      S("h1",  fontName=font_name+"-Bold", fontSize=18, leading=24,
                     textColor=DARK, spaceBefore=22, spaceAfter=4),
        "h2":      S("h2",  fontName=font_name+"-Bold", fontSize=13, leading=18,
                     textColor=DARK, spaceBefore=14, spaceAfter=3),
        "h3":      S("h3",  fontName=font_name+"-Bold", fontSize=11, leading=16,
                     textColor=DARK, spaceBefore=10, spaceAfter=2),
        "body":    S("body", alignment=TA_JUSTIFY),
        "bullet":  S("bullet", leftIndent=14, firstLineIndent=0),
        "num":     S("num",    leftIndent=14, firstLineIndent=0),
        "callout": S("callout", leftIndent=14, fontSize=10, leading=16, textColor=DARK),
        "code":    S("code", fontName="Courier", fontSize=8.5, leading=12,
                     leftIndent=12, textColor=DARK),
        "th":      S("th", fontName=font_name+"-Bold", fontSize=9, leading=13,
                     textColor=white, alignment=TA_CENTER),
        "td":      S("td", fontName=font_name, fontSize=9, leading=13,
                     textColor=DARK, alignment=TA_LEFT),
        "_accent": ACCENT,
        "_dark": DARK,
        "_muted": MUTED,
        "_bg_row": BG_ROW,
        "_bg_call": BG_CALL,
    }


def build_flowables(blocks: list, ST: dict, page_width: float, lm: float, rm: float) -> list:
    ACCENT = ST["_accent"]
    BG_CALL = ST["_bg_call"]
    story = []
    num_counter = 0
    content_width = page_width - lm - rm

    for btype, content in blocks:
        if btype == "h1":
            story.append(Spacer(1, 4))
            story.append(Paragraph(md_inline(content), ST["h1"]))
            story.append(HRFlowable(width="100%", thickness=1.5, color=ACCENT, spaceAfter=6))
            num_counter = 0
        elif btype == "h2":
            story.append(Paragraph(md_inline(content), ST["h2"]))
            num_counter = 0
        elif btype == "h3":
            story.append(Paragraph(md_inline(content), ST["h3"]))
        elif btype == "body":
            if content:
                story.append(Paragraph(md_inline(content), ST["body"]))
        elif btype == "bullet":
            story.append(Paragraph("• " + md_inline(content), ST["bullet"]))
        elif btype == "num":
            num_counter += 1
            story.append(Paragraph(f"{num_counter}. " + md_inline(content), ST["num"]))
        elif btype == "callout" and content:
            story.append(Spacer(1, 4))
            inner = Paragraph(md_inline(content), ST["callout"])
            t = Table([[inner]], colWidths=[content_width - 8])
            t.setStyle(TableStyle([
                ("BACKGROUND",   (0,0), (-1,-1), BG_CALL),
                ("LEFTPADDING",  (0,0), (-1,-1), 14),
                ("RIGHTPADDING", (0,0), (-1,-1), 10),
                ("TOPPADDING",   (0,0), (-1,-1), 8),
                ("BOTTOMPADDING",(0,0), (-1,-1), 8),
                ("LINEBEFORE",   (0,0), (0,-1), 4, ACCENT),
            ]))
            story.append(t)
            story.append(Spacer(1, 4))
        elif btype == "code":
            story.append(Spacer(1, 4))
            lines = content.split("\n")
            text = "<br/>".join(escape_xml(l) for l in lines)
            p = Paragraph(f'<font name="Courier" size="8">{text}</font>', ST["code"])
            t = Table([[p]], colWidths=[content_width - 8])
            t.setStyle(TableStyle([
                ("BACKGROUND",   (0,0), (-1,-1), HexColor("#F5F5F5")),
                ("LEFTPADDING",  (0,0), (-1,-1), 10),
                ("RIGHTPADDING", (0,0), (-1,-1), 10),
                ("TOPPADDING",   (0,0), (-1,-1), 8),
                ("BOTTOMPADDING",(0,0), (-1,-1), 8),
                ("LINEBEFORE",   (0,0), (0,-1), 3, ACCENT),
            ]))
            story.append(t)
            story.append(Spacer(1, 4))
        elif btype == "hr":
            story.append(HRFlowable(width="100%", thickness=0.5,
                                    color=HexColor("#CCCCCC"), spaceAfter=6))
        elif btype == "table":
            rows = content
            if not rows:
                continue
            col_count = max(len(r) for r in rows)
            col_w = content_width / col_count
            tdata = [[Paragraph(md_inline(c), ST["th"]) for c in rows[0]]]
            for row in rows[1:]:
                while len(row) < col_count:
                    row.append("")
                tdata.append([Paragraph(md_inline(c), ST["td"]) for c in row])
            t = Table(tdata, colWidths=[col_w] * col_count, repeatRows=1)
            t.setStyle(TableStyle([
                ("BACKGROUND",    (0,0), (-1,0),  ACCENT),
                ("TEXTCOLOR",     (0,0), (-1,0),  white),
                ("TOPPADDING",    (0,0), (-1,-1), 5),
                ("BOTTOMPADDING", (0,0), (-1,-1), 5),
                ("LEFTPADDING",   (0,0), (-1,-1), 6),
                ("RIGHTPADDING",  (0,0), (-1,-1), 6),
                ("ROWBACKGROUNDS",(0,1), (-1,-1), [white, ST["_bg_row"]]),
                ("BOX",           (0,0), (-1,-1), 0.5, HexColor("#CCCCCC")),
                ("VALIGN",        (0,0), (-1,-1), "TOP"),
            ]))
            story.append(Spacer(1, 6))
            story.append(t)
            story.append(Spacer(1, 6))

    return story


def make_cover_fn(style: dict, theme: dict, font_name: str):
    """Return an onPage callback that draws the cover on page 1."""
    ACCENT = HexColor(theme["accent"])
    title = style.get("cover_title", "")
    subtitle = style.get("cover_subtitle", "")
    meta = style.get("cover_meta", "")

    def draw_cover(canv, doc):
        if doc.page != 1:
            return
        pw, ph = doc.pagesize
        canv.saveState()
        canv.setFillColor(HexColor("#0F1F2E"))
        canv.rect(0, 0, pw, ph, fill=1, stroke=0)
        canv.setFillColor(ACCENT)
        canv.rect(0, ph - 8, pw, 8, fill=1, stroke=0)
        # dot grid
        canv.setFillColor(HexColor("#FFFFFF"))
        canv.setFillAlpha(0.06)
        for x in range(30, int(pw), 18):
            for y in range(30, int(ph), 18):
                canv.circle(x, y, 1, fill=1, stroke=0)
        canv.setFillAlpha(1)
        # title
        canv.setFillColor(white)
        canv.setFont(font_name + "-Bold", 32)
        canv.drawString(2.5*cm, ph * 0.62, title)
        if subtitle:
            canv.setFont(font_name, 18)
            canv.drawString(2.5*cm, ph * 0.55, subtitle)
        canv.setStrokeColor(ACCENT)
        canv.setLineWidth(2)
        canv.line(2.5*cm, ph * 0.52, pw - 2.5*cm, ph * 0.52)
        if meta:
            canv.setFillColor(HexColor("#607080"))
            canv.setFont(font_name, 10)
            canv.drawString(2.5*cm, 3.5*cm, meta)
        canv.restoreState()

    return draw_cover


def make_decorate_fn(style: dict, theme: dict, font_name: str, doc_title: str):
    """Return an onPage callback for header/footer on body pages."""
    ACCENT = HexColor(theme["accent"])
    MUTED = HexColor(theme["muted"])
    show_header = style.get("header", True)
    show_footer = style.get("footer", True)
    show_page_num = style.get("footer_page_number", True)

    def decorate(canv, doc):
        canv.saveState()
        lm, rm = doc.leftMargin, doc.rightMargin
        pw, ph = doc.pagesize
        top = ph - doc.topMargin
        if show_header:
            canv.setStrokeColor(ACCENT)
            canv.setLineWidth(1.5)
            canv.line(lm, top + 10, pw - rm, top + 10)
            canv.setFillColor(MUTED)
            canv.setFont(font_name, 7.5)
            canv.drawString(lm, top + 14, doc_title)
        if show_footer:
            canv.setStrokeColor(HexColor("#DDDDDD"))
            canv.setLineWidth(0.5)
            canv.line(lm, doc.bottomMargin - 10, pw - rm, doc.bottomMargin - 10)
            canv.setFillColor(MUTED)
            canv.setFont(font_name, 7.5)
            if show_page_num:
                canv.drawRightString(pw - rm, doc.bottomMargin - 20, str(doc.page))
        canv.restoreState()

    return decorate


def build_pdf(md_text: str, output_path: str, font_path: str,
              theme: dict, style: dict, doc_title: str) -> None:
    """Build PDF from markdown text and write to output_path."""
    font_name = register_font(font_path)
    ST = make_styles(font_name, theme)

    page_size = LETTER if style.get("page_size", "A4").upper() == "LETTER" else A4
    margin = style.get("margin_cm", 2.5) * cm
    extra = 0.8 * cm  # extra for header/footer

    doc = BaseDocTemplate(
        output_path,
        pagesize=page_size,
        leftMargin=margin, rightMargin=margin,
        topMargin=margin + extra, bottomMargin=margin + extra,
        title=doc_title,
    )

    pw = page_size[0]
    frame = Frame(doc.leftMargin, doc.bottomMargin, doc.width, doc.height, id="body")

    has_cover = style.get("cover", True)

    if has_cover:
        cover_fn = make_cover_fn(style, theme, font_name)
        body_fn = make_decorate_fn(style, theme, font_name, doc_title)

        def on_page(canv, doc):
            if doc.page == 1:
                cover_fn(canv, doc)
            else:
                body_fn(canv, doc)
    else:
        body_fn = make_decorate_fn(style, theme, font_name, doc_title)
        on_page = body_fn

    tmpl = PageTemplate(id="main", frames=[frame], onPage=on_page)
    doc.addPageTemplates([tmpl])

    blocks = parse_md(md_text)
    story = []
    if has_cover:
        story.append(PageBreak())  # blank cover page
    story += build_flowables(blocks, ST, pw, margin, margin)
    doc.build(story)
